#include <stdio.h> 
#include <stdlib.h>
#include <pthread.h> //crear funciones
#include <unistd.h> // libreria para sleep 
#include<windows.h>
#include <string.h>
#include <time.h>
void *verde(void*); 
void *naranja(void*); 
void *rojo(void*);

void *verde(void *tid)
{
system("cls"); //cls o clear es para limpiar la pantalla de comandos 
system("color 20");

}
void *naranja(void *tid)
{
system("cls"); 
system("color E0"); 

}
void *rojo(void *tid)
{
system("cls"); 
system("color 40"); 

}
int main()
{
int i = 1;
int tiempo,tiempo1,tiempo2;
printf("Introduzca el tiempo en segundos del color verde: ");
scanf("%d",&tiempo);
printf("Introduzca el tiempo en segundos del color amarillo: ");
scanf("%d",&tiempo1);
printf("Introduzca el tiempo en segundos del color rojo: ");
scanf("%d",&tiempo2);

pthread_t pthread_rojo, pthread_naranja, pthread_verde;

pthread_create (&pthread_verde, NULL, verde, (void *)i); 
pthread_join(pthread_verde, NULL);
sleep(tiempo);

pthread_create (&pthread_naranja,NULL, naranja, (void *)i); 
pthread_join(pthread_naranja, NULL);
sleep(tiempo1);

pthread_create (&pthread_rojo, NULL, rojo, (void *)i); 
pthread_join(pthread_rojo, NULL);
sleep(tiempo2);

pthread_exit(NULL);

return 0;
}
